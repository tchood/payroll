def age(as_at = Time.now)
raise 'Date expected in age method' unless as_at.is_a?(Time)
  as_at.year - @birthdate.year - ((as_at.month > @birthdate.month || (as_at.month == @birthdate.month && as_at.day >= @birthdate.day)) ? 0 : 1)
end
